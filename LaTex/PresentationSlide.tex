\section{Introduction}

\subsection{Motivation}
\begin{frame}{Motivation}
\begin{itemize}
  \item If an edge can be found using publicly available data, then there are some grounds to partially reject the \textbf{Efficient Market Hypothesis}.
  \item Obtaining an edge in forecasting the market direction allows for traders to make more informed decisions about the economy and how they should position thier trade and order sizes.
\end{itemize}
\end{frame}

\subsection{Research Question}
\begin{frame}{Research Question}
\begin{itemize}
  \item Goal: Forecasting the \textbf{direction} of the S\&P 500 Index.
  \item Research Problem: Is there information in price and volume that can be used to forecast market direction.
  \item Modeling approach: Perform feature engineering and selection/dimensionality reduction on raw data; fit logistic regression, random forest classification and support vector machine models with optimal hyperparameter values obtained through iterative selection. %
  %Compare traditional statistical models (e.g., logistic regression) with modern statistical learning techniques (e.g., random forests, SVM, neural networks) to evaluate their predictive performance and implications for market efficiency
\end{itemize}
\end{frame}
\subsection{Background}
\begin{frame}{Background}
\begin{itemize}
  \item \textbf{Weak-form EMH}: past prices/volume should not predict future returns; model performance tests weak-form efficiency.
  \item The \textbf{S\&P 500} index tracks the top 500 US companies by their \textbf{float-adjusted market capitalization (FMC)} and serves as the leading benchmark for the health of the US economy.
  % \item Index values are computed using \textbf{float-adjusted market capitalization (FMC)}, so larger companies have a proportionally greater impact on the index
  \item Forecasting the direction of the S\&P 500 Index is \textbf{complex} %the index point value is influenced by individual company performance, economic indicators, geopolitical events, and investor sentiment. Comment so that we talk about it but not too much on the slide
  \item The data utilized is purely \textbf{historical} prices and trading volume. %when markets close at the end of the day, it already reflects economic and political condition of that day while concurrent indicators are typically unavailable at prediction time
conditions,
\end{itemize}
\end{frame}






\section{Dataset \& preprocessing}
\subsection{Dataset \& preprocessing}
\begin{frame}{Data collection and preprocessing}
\begin{itemize}
  \item Raw data: retrieved through Yahoo Finance's API (\texttt{yfinance}). 
  \begin{itemize}
    \item Contains open, close, high, low prices, and traded volume metrics (OHLCV features) on stocks listed on the S\&P 500 index
    \item From \textbf{January 1, 2018} to \textbf{December 31, 2025}
  \end{itemize}
  \item Cleaning steps: ticker changes (mergers), invalid/frozen-state entries, market holidays, and delisted/newly-listed stocks. Drop missing-value stocks.
  \item Dimensionality: \textbf{2010 trading days} $\times$ \textbf{2,520 features}.
\end{itemize}
\end{frame}

\subsection{Feature Engineering}
\begin{frame}{Feature Engineering and target variable}
\begin{itemize}
  \item Target variable: \textbf{binary} indicator (whether the S\&P 500 index will go up (1) or down (0) the next day, based on the closing value)
  \item Predictors is extended with:
  \begin{itemize}
    \item \textbf{Lag values} --- capture momentum and time-series structure.
    \item \textbf{Day-to-day \% change} \& \textbf{daily range} --- relative price movement and intraday volatility.
    \item \textbf{Volatility} \& \textbf{rolling z-score} --- variability and outlier/event detection.
    \item \textbf{Max/min channel position} --- reflect breakout signals.
    \item \textbf{Rolling VWAP} \& \textbf{day of week} --- volume-weighted context and seasonlity.
  \end{itemize}
  \item \textbf{Non-stationary} features (OP, CP, HP, LP) are \textbf{removed} after engineering.
\end{itemize}
\end{frame}

\begin{frame}{Methods}
\begin{itemize}
  \item Data Characteristics: 
  \begin{itemize}
    \item \textbf{many features} $\to$ Dimensionality reduction: (PCA), regularization (Ridge, LASSO),

  \item Models used based on data nature
  \begin{itemize}
    \item Logistic Regression
    \item Random Forest Classification
    \item Support Vector Machines
  \end{itemize}
  \item Model selection (Hyperparameter selection also)
  \begin{itemize}
    \item combination of available metric for different methods test accuracy, F1, ROC-AUC
    \item the more \textbf{parsimonious} one is preferred when similar performances happen
    \item cross validation: k-fold and Walk-forward
  \end{itemize}
\end{itemize}

\section{Limitations \& next steps}
\begin{frame}{Limitation}
  %Prof's comment: too much matter clustered on one page. summarize this
  \begin{itemize}
    \item Data limitations: only historical price and volume data, no sentiment analysis, macroeconomic indicators, or alternative data sources
    \item Model limitations: did not explore mix model methods, more complicated deep learning architectures
    \item Evaluation limitations: only used accuracy, F1-score, and ROC-AUC mainly based on wark-forward validation and cross validation. Other rules may be applied in extension of this project (1SE rule)
    \item Time constraints: limited time for extensive hyperparameter tuning and model experimentation
    \item Resource constraints: computational resources limited the ability to train more complex models or perform extensive cross-validation
    \end{itemize}
\end{frame}

\begin{frame}{Next steps}
  %Prof's comment: too much matter clustered on one page. summarize this
\begin{itemize}
  \item Conduct more extensive hyperparameter tuning and model experimentation
  \item Implement a more comprehensive evaluation framework, including additional metrics and validation techniques
  \item Include more political and economic indicators, sentiment analysis, and alternative data sources to enhance predictive power
  \item Explore more sophisticated models (more complicated deep learning architectures) and some combination with economic method of time series forecasting (e.g., ARIMA, GARCH)
\end{itemize}
\end{frame}

\section{Final summary}
\begin{frame}{Summary}
  %Prof's comment: too much matter clustered on one page. summarize this
\begin{itemize}
  \item Predicting the direction of the stock market is a complex task with many influencing factors. 
  \item While some models can achieve slightly better than random performance, there is still significant room for improvement. 
  \item Relying solely on historical daily price and volume data may not be sufficient for making informed trading decisions
  \item Better result for short time windows suggests implications for high-frequency trading strategies and intraday trading.
  \item Future work could focus on short-time dependencies, more sophisticated models, additional features (e.g., sentiment analysis, macroeconomic indicators), and alternative modeling approaches (e.g., ensemble methods, deep learning architectures).
  \end{itemize}
\end{frame}
