\section{Introduction}

\subsection{Motivation}
\begin{frame}{Motivation}
\begin{itemize}
    \item Predicting market direction is important for practical trading decisions.
    \item Prediction performance itself also contains important information of historical signals and market direction.
    \item The result will shed light on the plaussibility of Efficient Market Theory in practice.
    \item If historical signals are predictive out of sample, this challenges weak-form market efficiency.
    \item If not, the results are more consistent with the \textbf{Efficient Market Hypothesis}.

\end{itemize}
\end{frame}

\subsection{Research Question}
\begin{frame}{Research Question}
\begin{itemize}
  \item Goal: Forecasting the \textbf{direction} of the S\&P 500 Index (market direction)
  \item Research Problem: Is there historical information in price and volume that can be used to forecast market direction.
  \item Modeling approach: 
  \begin{itemize}
    \item Perform feature engineering and selection/dimensionality reduction
    \item Fit models suitable to data main characteristics
    \item Select optimal hyperparameter values through iterative selection. %
  \end{itemize}
\end{frame}
\subsection{Background}
\begin{frame}{Background}
\begin{itemize}
  \item \textbf{Weak-form EMH}: past prices/volume should not predict future returns; model performance tests weak-form efficiency.
  \item The \textbf{S\&P 500} index tracks the top 500 US companies by their \textbf{float-adjusted market capitalization (FMC)} and serves as the leading benchmark for the health of the US economy.
  % \item Index values are computed using \textbf{float-adjusted market capitalization (FMC)}, so larger companies have a proportionally greater impact on the index
  \item Forecasting the direction of the S\&P 500 Index is \textbf{complex} %the index point value is influenced by individual company performance, economic indicators, geopolitical events, and investor sentiment. Comment so that we talk about it but not too much on the slide
  \item The data utilized is purely \textbf{historical} prices and trading volume. %when markets close at the end of the day, it already reflects economic and political condition of that day while concurrent indicators are typically unavailable at prediction time
conditions,
\end{itemize}
\end{frame}






\section{Dataset \& preprocessing}
\subsection{Dataset \& preprocessing}
\begin{frame}{Data collection and preprocessing}
\begin{itemize}
  \item Raw data: retrieved through Yahoo Finance's API (\texttt{yfinance}). 
  \begin{itemize}
    \item Contains open price (OP), close price (CP), high price (HP), low price (LP), and trading volume (V). They are called OHLCV features) on stocks listed on the S\&P 500 index
    \item From \textbf{January 1, 2018} to \textbf{December 31, 2025}
  \end{itemize}
  \item Cleaning steps: ticker changes (mergers), invalid/frozen-state entries, market holidays, and delisted/newly-listed stocks. Drop missing-value stocks.
  \item Dimensionality: \textbf{2009 trading days} $\times$ \textbf{2,520 features}.% Comment: update later
\end{itemize}
\end{frame}

\subsection{Feature Engineering}
\begin{frame}{Feature Engineering and target variable}
\begin{itemize}
  \item Target variable: \textbf{binary} indicator (whether the S\&P 500 index will go up (1) or down (0) the next day, based on the closing value)
  \item Predictors is extended with:
  \begin{itemize}
    \item \textbf{Lag values} --- capture momentum and time-series structure.
    \item \textbf{Day-to-day \% change} \& \textbf{daily range} --- relative price movement and intraday volatility.
    \item \textbf{Volatility} \& \textbf{rolling z-score} --- variability and outlier/event detection.
    \item \textbf{Max/min channel position} --- reflect breakout signals.
    \item \textbf{Rolling VWAP} \& \textbf{day of week} --- volume-weighted context and seasonlity.
  \end{itemize}
  \item \textbf{Non-stationary} features (OP, CP, HP, LP) are \textbf{removed} after engineering.
\end{itemize}
\end{frame}

\section{Methods}
% Slide 1
\begin{frame}[t,shrink=8]{Method Overview: Data Challenges and solutions}

\begin{block}{High-Dimensional Predictors ($p$ large relative to $n$)}
\begin{itemize}
    \item \textbf{Principal Component Analysis (PCA):} reduce predictors to M principal components (Ch. 14, Section 14.5.1, ESL)
    % ESL reference:
    % Chapter 14, "Unsupervised Learning", Section 14.5.1 "Principal Components", p. 534.
    % The notation below is a slide-friendly score representation:
    % z_{im} = \sum_{j=1}^p x_{ij}\phi_{jm},
    % where \phi_m is the mth loading direction.
    % This is consistent with the principal-component construction in ESL Ch. 14, §14.5.1.
    \[
    z_{im} = \sum_{j=1}^p x_{ij} \phi_{jm}, \qquad m=1,\dots,M
    \]
    where $\phi_m$ is the largest $m$th loading direction.
    
    \item \textbf{Regularization: LASSO} (Ch. 3, Section 3.4.2, ESL)
    % ESL reference for lasso:
    % Chapter 3, Section 3.4.2 "The Lasso", starts p. 68.
    % More generally, ESL Eq. (3.53) in §3.4.3 gives the bridge penalty, p. 72;
    % q=1 corresponds to the lasso and q=2 to ridge.
    % The criterion below is the lasso specialization used on the slide.
    \[
    \hat{\beta}^{\mathrm{lasso}}
    = \arg\min_{\beta}\left\{
    \sum_{i=1}^n (y_i-\beta_0-x_i^\top \beta)^2
    + \lambda \sum_{j=1}^p |\beta_j|
    \right\}
    \]
\end{itemize}
\end{block}

\begin{block}{Multicollinearity}
\begin{itemize}
    \item  Regularization: Ridge and LASSO % ESL reference for ridge:
    % Chapter 3, "Linear Methods for Regression", Section 3.4.1 "Ridge Regression", starts p. 61.
    % Penalized least-squares ridge criterion appears as Eq. (3.43), p. 64.
    \[
    \hat{\beta}^{\mathrm{ridge}}
    = \arg\min_{\beta}\left\{
    \sum_{i=1}^n (y_i-\beta_0-x_i^\top \beta)^2
    + \lambda \sum_{j=1}^p \beta_j^2
    \right\}
    \]
    \item Sector-level feature aggregation
    \item PCA 
    % ESL reference for PCA:
    % Chapter 14, Section 14.5.1 "Principal Components", p. 534.
\end{itemize}
\end{block}

\begin{alertblock}{Time-Series Structure}
\begin{itemize}
    \item Lagged predictors preserve temporal ordering
    \item Models with inherent time-series nature
    %\item Walk-forward validation respects time dependence
    % ESL reference:
    % Chapter 7, "Model Assessment and Selection", starts p. 219.
    % k-fold cross-validation is discussed in this chapter; walk-forward validation
    % is a time-series adaptation of the same model-assessment principle.
\end{itemize}
\end{alertblock}

\vfill
{\tiny \textit{Source:} Hastie, Tibshirani, and Friedman (2009), \textit{The Elements of Statistical Learning}, Ch.~3, Ch.~7, and Ch.~14.}

\end{frame}


\begin{frame}[t,shrink=8]{Method Overview: Models and Model Selection}

\begin{block}{Classification Models}
\begin{itemize}
    \item \textbf{Logistic Regression}
    % ESL reference:
    % Chapter 4, Section 4.4, p. 119.
    % ESL Eq. (4.17) gives the multiclass log-odds formulation. Use binary logistic special case rather than original formula in the book
    \[
    \Pr(G=1\mid X=x)
    =
    \frac{\exp(\beta_0+x^\top\beta)}
    {1+\exp(\beta_0+x^\top\beta)}
    \]
    Easy to be applied with Ridge and LASSO
    \item \textbf{Random Forest}: No explicit restriction for large number of features.
    \begin{itemize}
        \item Ensemble of classification trees
        \item Built on bootstrap samples
        \item Final prediction by majority vote
        % ESL reference:
        % Chapter 15, "Random Forests".
        % Use as a conceptual chapter citation rather than a specific displayed formula.
    \end{itemize}
    
    \item \textbf{Support Vector Machine}: No restriction of number of features, non-linear separable
    % ESL reference:
    % Chapter 12, Section 12.2 p. 417. for SVC and Eqs. (12.7)--(12.8), pp. 420--421. for SVM
    \[
    \min_{\beta,\beta_0,\xi}\ 
    \frac{1}{2}\|\beta\|^2 + C\sum_{i=1}^N \xi_i
    \]
    \[
    \text{subject to } \quad
    y_i(x_i^\top\beta+\beta_0)\ge 1-\xi_i,\qquad
    \xi_i\ge 0,\quad i=1,\dots,N
    \]

    \item Experiment with some neural net models (CNN-2D, LSTM, RNN)
    
\end{itemize}
\end{block}

\begin{block}{Model Tuning and Selection} (Ch. 7, ESL)
\begin{itemize}
    \item Hyperparameters selected by cross-validation
    \item Validation methods:
    \begin{itemize}
        \item $k$-fold cross-validation
            % ESL reference:
            % Chapter 7, "Model Assessment and Selection", p. 219.
        %\item Walk-forward validation
    \end{itemize}
    \item Evaluation metrics: Accuracy, F1-score, ROC-AUC
    \item 1SE rule: Prefer the more \textbf{parsimonious} model when performance is similar
          % Chapter 7, "Model Assessment and Selection", p. 219.
\end{itemize}
\end{block}

\vfill
{\tiny \textit{Source:} Hastie, Tibshirani, and Friedman (2009), \textit{The Elements of Statistical Learning}, Ch.~4, Ch.~7, Ch.~12, and Ch.~15.}

\end{frame}

\section{Limitations \& next steps}
\begin{frame}{Limitation}
  %Prof's comment: too much matter clustered on one page. summarize this
  \begin{itemize}
    \item Data: 8 years rather than longer, causing mild response bias
    \item Models: Limited grid-search lists in main models, too simple neural net architectures in experiment one.
    \item Evaluation: Hard to use multiple metrics when they are not available for all models
    \item Time constraints: limited time for extensive hyperparameter tuning 
    \item Computational resources: limited in training more complex models
    \end{itemize}
\end{frame}

\begin{frame}{Next steps}
  %Prof's comment: too much matter clustered on one page. summarize this
\begin{itemize}
  \item Conduct more extensive hyperparameter tuning, widen grid search list
  \item Include longer time series, shorter time window
  \item Explore more sophisticated models (more complicated deep learning architectures)
  \item Potential of combination with economic method of time series forecasting (e.g., ARIMA, GARCH)
\end{itemize}
\end{frame}

\section{Final summary}
\begin{frame}{Summary}
  %Prof's comment: too much matter clustered on one page. summarize this
\begin{itemize}
  \item Supporting evidence for: market is quite efficient when looking overally at daily trading, but not efficient in shorter time window.
  \item Relying solely on historical daily price and volume data may not be sufficient. 
  \item Better result for short time windows suggests implications for high-frequency trading strategies and intraday trading.
  \item Future work could focus on short-time dependencies, more sophisticated models, additional features (e.g., sentiment analysis, macroeconomic indicators), and alternative modeling approaches (e.g., ensemble methods, deep learning architectures).
  \end{itemize}
\end{frame}
